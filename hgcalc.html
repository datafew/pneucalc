Public Function BlowerMotorGuc(ByVal Model As String, ByVal n As Double, ByVal P_kPa As Double) As Double
    Dim La As Double, req As Double
    Dim std As Variant, i As Long
    La = BlowerGuc(Model, n, P_kPa)
    If La < 0 Then: BlowerMotorGuc = -1: Exit Function
    req = La * 1.15
    std = Array(0.75, 1.1, 1.5, 2.2, 3, 4, 5.5, 7.5, 11, 15, 18.5, 22, 30, 37, 45, 55, 75, 90, 110, 132, 160, 185, 200, 220, 250, 280, 315)
    For i = LBound(std) To UBound(std)
        If std(i) >= req Then
            BlowerMotorGuc = std(i)
            Exit Function
        End If
    Next i
    BlowerMotorGuc = std(UBound(std))
End Function

    @classmethod
    def calculate(cls, model: str, n: float, p_kpa: float, safety_factor: float = 1.15):
        m_key = model.upper().strip()
        if m_key not in cls.COEFFICIENTS:
            raise ValueError(f"Model '{model}' not found.")
        c = cls.COEFFICIENTS[m_key]
        q_th = (c["v_rev"] * n) / 1000.0
        q_slip = c["a"] * (p_kpa ** c["b"])
        qs = max(0.0, q_th - q_slip)
        la = c["c0"] * n * p_kpa + c["c1"] * n
        
        req = la * safety_factor
        po = cls.STANDARD_MOTORS[-1]
        for std in cls.STANDARD_MOTORS:
            if std >= req:
                po = std
                break
        return {
            "Qth_m3_min": q_th,

            "La_kW": la,
            "Po_kW": po
